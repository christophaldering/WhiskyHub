### Problem

### Lösung

**Datei: `client/src/labs/pages/LabsDistilleries.tsx`**

```tsx
const [viewMode, setViewMode] = useState<'list' | 'map'>('list');

// In der Render-Funktion:
<div className="labs-segmented" style={{ marginBottom: '1rem' }}>
  <button
    className={`labs-segmented-btn ${viewMode === 'list' ? 'active' : ''}`}
    onClick={() => setViewMode('list')}
  >
    {t('distilleries.listView')}  {/* "Liste" / "List" */}
  </button>
  <button
    className={`labs-segmented-btn ${viewMode === 'map' ? 'active' : ''}`}
    onClick={() => setViewMode('map')}
  >
    {t('distilleries.mapView')}   {/* "Karte" / "Map" */}
  </button>
</div>

{viewMode === 'list' && <DistilleriesList distilleries={filtered} />}
{viewMode === 'map'  && <DistilleriesMap  distilleries={filtered} />}
```

#### 4b — Map-Komponente (Leaflet oder SVG-basiert):

Falls Leaflet bereits installiert (`leaflet` in package.json), verwende:

```tsx
import { MapContainer, TileLayer, Marker, Popup } from 'react-leaflet';

function DistilleriesMap({ distilleries }: { distilleries: Distillery[] }) {
  return (
    <div style={{ height: '60vh', borderRadius: '12px', overflow: 'hidden', border: '0.5px solid var(--labs-border)' }}>
      <MapContainer
        center={[56.5, -4.5]}  // Schottland als Default
        zoom={5}
        style={{ height: '100%', width: '100%' }}
      >
        <TileLayer
          url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png"
          attribution='© OpenStreetMap contributors'
        />
        {distilleries
          .filter(d => d.latitude && d.longitude)
          .map(d => (
            <Marker key={d.id} position={[d.latitude!, d.longitude!]}>
              <Popup>
                <strong>{d.name}</strong><br/>
                {d.region}, {d.country}
              </Popup>
            </Marker>
          ))
        }
      </MapContainer>
    </div>
  );
}
```

Falls Leaflet NICHT installiert ist:
```bash
npm install leaflet react-leaflet @types/leaflet
```

Und CSS-Import in `index.html` oder Entry-File:
```html
<link rel="stylesheet" href="https://unpkg.com/leaflet@1.9.4/dist/leaflet.css"/>
```

import { useTranslation } from 'react-i18next';
import { useLocation } from 'wouter';

export default function LabsEntdecken() {
  const { t } = useTranslation();
  const [, navigate] = useLocation();

  const whiskyItems = [
    { label: t('discover.allWhiskies'), sub: t('discover.allWhiskiesSub'), path: '/labs/explore' },
    { label: t('discover.compare'),     sub: t('discover.compareSub'),     path: '/labs/taste/compare' },

    { label: t('discover.lexicon'),       sub: t('discover.lexiconSub'),       path: '/labs/discover/lexicon' },

    { label: t('discover.bottlers'),      sub: t('discover.bottlersSub'),      path: '/labs/discover/bottlers' },
    { label: t('discover.guide'),         sub: t('discover.guideSub'),         path: '/labs/discover/guide' },
    { label: t('discover.templates'),     sub: t('discover.templatesSub'),     path: '/labs/discover/templates' },
    { label: t('discover.vocabulary'),    sub: t('discover.vocabularySub'),    path: '/labs/discover/vocabulary' },

    { label: t('discover.rabbitHole'),    sub: t('discover.rabbitHoleSub'),    path: '/labs/discover/rabbit-hole' },
  ];

  const renderItems = (items: typeof whiskyItems) => (
    <div className="labs-grouped-list">
      {items.map(item => (
        <div
          key={item.path}
          className="labs-list-row"
          onClick={() => navigate(item.path)}
          style={{ cursor: 'pointer' }}
        >
          <div style={{ flex: 1 }}>

              {item.sub}

        </div>

  );

  return (
    <div style={{ padding: '1.5rem 1.25rem 6rem', maxWidth: '480px', margin: '0 auto' }}>
      <p style={{ fontFamily: "'DM Sans', sans-serif", fontSize: '11px', textTransform: 'uppercase',

        {t('discover.sub')}
      </p>

      <p style={{ fontFamily: "'DM Sans', sans-serif", fontSize: '11px', textTransform: 'uppercase',
