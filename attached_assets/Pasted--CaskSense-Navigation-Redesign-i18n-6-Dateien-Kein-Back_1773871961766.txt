#### Nav-Container und Item-Rendering ersetzen:

```tsx
<nav style={{
  position: 'fixed',
  bottom: 0,
  left: 0,
  right: 0,
  display: 'flex',
  alignItems: 'flex-start',
  justifyContent: 'space-around',
  borderTop: '0.5px solid var(--labs-border, rgba(255,255,255,0.12))',
  backdropFilter: 'blur(20px)',
  WebkitBackdropFilter: 'blur(20px)',
  backgroundColor: 'var(--labs-nav-bg, rgba(18,14,10,0.94))',
  paddingBottom: 'env(safe-area-inset-bottom, 8px)',
  paddingTop: '6px',
  zIndex: 100,
}}>
  {navItems.map((item) => {
    const isActive = item.matchPaths.some(p => location.startsWith(p));
    return (
      <Link
        key={item.pathPrefix}
        href={item.pathPrefix}
        style={{
          display: 'flex',
          flexDirection: 'column',
          alignItems: 'center',
          gap: '3px',
          padding: '6px 0 4px',
          flex: 1,
          textDecoration: 'none',
          color: isActive
            ? 'var(--labs-gold, #bf8f3e)'
            : 'var(--labs-text-muted, rgba(255,255,255,0.45))',
          transition: 'color 0.15s ease',
        }}
      >
        {item.icon(isActive)}
        <span style={{
          fontSize: '10px',
          fontFamily: isActive
            ? 'var(--font-display, "Playfair Display", serif)'
            : 'var(--font-ui, "DM Sans", sans-serif)',
          fontWeight: isActive ? 500 : 400,
          letterSpacing: '0.02em',
          lineHeight: 1,
        }}>
          {t(item.labelKey)}
        </span>
      </Link>
    );
  })}
</nav>
```
