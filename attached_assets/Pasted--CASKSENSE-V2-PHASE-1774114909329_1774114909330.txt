  export interface TastingConfig {
    name: string
    date: string
    time: string
    location: string
    format: "blind" | "open"
    scale: "100" | "20" | "10"

  export interface WhiskyEntry {
    id?: string
    name: string
    distillery?: string
    region?: string
    cask?: string
    age?: number
    abv?: number
    flavorProfile?: string
  }

  export interface TastingData {
    id: string
    code: string
    name: string
    status: "draft" | "open" | "closed" | "reveal" | "archived"
    format: "blind" | "open"

    createdAt: string
  }

  export interface ParticipantData {
    id: string
    name: string
    ratingStatus: "all" | "partial" | "none"
    source: "app" | "paper"
  }
