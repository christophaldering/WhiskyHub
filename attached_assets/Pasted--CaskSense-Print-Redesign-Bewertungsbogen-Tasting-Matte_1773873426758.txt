# CaskSense — Print Redesign
## Bewertungsbogen · Tasting-Matte · Menükarte
## Datei: client/src/components/printable-tasting-sheets.tsx

---

## Überblick

Drei Druckdokumente werden überarbeitet. Alle drei teilen dieselben Designprinzipien:

**Printer-friendly zuerst** — kein dunkler Hintergrund, keine Vollfarb-Blöcke.
Nur goldene Linien, Rahmen und Akzente. Druckt auf jedem Drucker sauber,
auch schwarzweiß.

**Typografie-System** — konsistent mit der App:
- Whisky-Namen: Playfair Display, kursiv, immer vollständig, nie abgeschnitten
- Labels/Kategorien: DM Sans, Uppercase, letter-spacing
- Body: DM Sans, normal

**Persönlicher QR-Code** — jeder Bogen gehört jemandem.

**Korrekte Dimensionen** — Nase · Gaumen · Abgang · Gesamt (4 Dimensionen).
Balance wurde entfernt. Nicht wieder einfügen.

**Flavour-Tags** — NUR wenn `whisky.flavorProfile` gesetzt ist. Nie erfinden.
Aus dem flavorProfile werden max. 3 Tags abgeleitet. Ist kein Profil gesetzt:
keine Tags, kein leerer Bereich.

```typescript
const PRINT_COLORS = {
  gold:        '#8b6914',          // Primärakzent — Zahlen, Labels, Linien
  goldLight:   'rgba(139,105,20,0.2)',  // Trennlinien zwischen Whiskies
  goldMid:     'rgba(139,105,20,0.4)', // Haupttrennlinien, Rahmen
  goldBorder:  'rgba(139,105,20,0.6)', // Score-Kreise, Checkboxen
  black:       '#1a1208',          // Überschriften, Whisky-Namen
  textDark:    '#2d2010',          // Body-Text
  textMuted:   '#888070',          // Metadaten, Untertitel
  white:       '#ffffff',          // Hintergrund — immer weiß
};
// Kein dunkler Hintergrund. Keine Vollfarb-Blöcke. Ausnahme: Cover-Seite
// der Menükarte darf einen sehr hellen Rahmen (2pt goldMid) haben.
```

```typescript
// ── QR-Code pro Teilnehmer ──────────────────────────────────────────────
// Generiert einen individuellen QR-Code der zu
// /labs/tastings/:tastingId/scan?participant=:participantId führt.
// Verwendet die bestehende qrcode-Library.

import QRCode from 'qrcode';

async function generateParticipantQR(
  tastingId: string,
  participantId: string,
  doc: jsPDF,
  x: number,
  y: number,
  size: number = 20  // mm
): Promise<void> {
  const url = `${window.location.origin}/labs/tastings/${tastingId}/scan?participant=${participantId}`;
  try {
    const dataUrl = await QRCode.toDataURL(url, {
      width: 200,
      margin: 1,
      color: { dark: '#1a1208', light: '#ffffff' },
    });
    doc.addImage(dataUrl, 'PNG', x, y, size, size);
    doc.setFontSize(6);
    doc.setTextColor(PRINT_COLORS.textMuted);
    doc.text(participantName, x + size / 2, y + size + 3, { align: 'center' });
  } catch (e) {
    // QR-Fehler still ignorieren — Dokument trotzdem generieren
    console.warn('[Print] QR generation failed:', e);
  }
}

// ── Flavour-Tags aus flavorProfile ──────────────────────────────────────
// Leitet max. 3 lesbare Tags aus dem flavorProfile-String ab.
// Gibt leeres Array zurück wenn kein Profil gesetzt.

function getFlavorTags(whisky: Whisky): string[] {
  if (!whisky.flavorProfile) return [];
  const profileMap: Record<string, string[]> = {
    'sherried-rich':      ['Sherry', 'Trockenfrüchte', 'Würze'],
    'bourbon-classic':    ['Vanille', 'Karamell', 'Eiche'],
    'peated-maritime':    ['Rauch', 'Salz', 'Torf'],
    'highland-elegant':   ['Honig', 'Blumen', 'Frucht'],
    'speyside-fruity':    ['Apfel', 'Birne', 'Malz'],
    'island-coastal':     ['Meeresluft', 'Heide', 'Gewürz'],
  };
  return profileMap[whisky.flavorProfile] ?? [];
}

// ── Score-Kreise zeichnen ────────────────────────────────────────────────
// Zeichnet 10 nummerierte Kreise (1-10) in einer Zeile.
// Printer-friendly: nur Kontur, kein Farbfill.

function drawScoreCircles(
  doc: jsPDF,
  x: number,
  y: number,
  circleSize: number = 5  // mm Durchmesser
): void {
  for (let i = 1; i <= 10; i++) {
    const cx = x + (i - 1) * (circleSize + 1);
    doc.setDrawColor(PRINT_COLORS.gold);
    doc.setLineWidth(0.3);
    doc.circle(cx + circleSize / 2, y + circleSize / 2, circleSize / 2, 'S');
    doc.setFontSize(6);
    doc.setTextColor(PRINT_COLORS.gold);
    doc.text(String(i), cx + circleSize / 2, y + circleSize / 2 + 1.5, { align: 'center' });
  }
}

// ── Goldene Trennlinie ───────────────────────────────────────────────────
function drawGoldLine(doc: jsPDF, x: number, y: number, width: number, opacity: number = 0.3): void {
  doc.setDrawColor(139, 105, 20);
  doc.setLineWidth(0.3);
  doc.setGState(doc.GState({ opacity }));
  doc.line(x, y, x + width, y);
  doc.setGState(doc.GState({ opacity: 1 }));
}
```

## SCHRITT 2 — Bewertungsbogen: `drawScoringPage()` ersetzen

### Vollständig neue Implementierung:

```typescript
async function drawScoringPage(
  doc: jsPDF,
  tasting: Tasting,
  whiskies: Whisky[],
  participant: Participant,
  pageWidth: number,
  pageHeight: number,
  margin: number = 15
): Promise<void> {
  const usableWidth = pageWidth - 2 * margin;
  let y = margin;

  // ── HEADER ─────────────────────────────────────────────────────────────
  // Tasting-Name links, QR-Code + Teilnehmer rechts
  // Kein dunkler Hintergrund — nur goldene Trennlinie darunter

  // Label
  doc.setFontSize(7);
  doc.setTextColor(PRINT_COLORS.gold);
  doc.setFont('helvetica', 'normal');
  doc.text('BEWERTUNGSBOGEN', margin, y + 3);

  // Tasting-Name
  doc.setFontSize(14);
  doc.setTextColor(PRINT_COLORS.black);
  doc.setFont('helvetica', 'italic');
  doc.text(tasting.title, margin, y + 10);

  // Datum + Host
  doc.setFontSize(7);
  doc.setTextColor(PRINT_COLORS.textMuted);
  doc.setFont('helvetica', 'normal');
  const meta = [
    tasting.date ? formatDate(tasting.date) : '',
    tasting.hostName ?? '',
  ].filter(Boolean).join(' · ');
  doc.text(meta, margin, y + 15);

  // QR-Code oben rechts (20x20mm)
  const qrX = pageWidth - margin - 22;
  await generateParticipantQR(tasting.id, participant.id, doc, qrX, y, 20);

  // Goldene Linie unter dem Header
  y += 22;
  drawGoldLine(doc, margin, y, usableWidth, 0.4);
  y += 5;

  // ── WHISKY-BLÖCKE ──────────────────────────────────────────────────────
  const DIMS = ['Nase', 'Gaumen', 'Abgang', 'Gesamt'] as const;
  // WICHTIG: Balance ist entfernt. Nur diese 4 Dimensionen.

  const whiskyChunkSize = tasting.isLandscape ? 6 : 4;
  // Max. Whiskies pro Seite — bei Überlauf neue Seite

  for (let i = 0; i < whiskies.length; i++) {
    const whisky = whiskies[i];

    // Seitenumbruch prüfen
    if (y > pageHeight - 40) {
      doc.addPage();
      y = margin;
    }

    // Nummer + Name
    doc.setFontSize(16);
    doc.setTextColor(PRINT_COLORS.gold);
    doc.setFont('helvetica', 'normal');
    doc.text(String(i + 1), margin, y + 6);

    // Whisky-Name — kursiv, nie abschneiden
    const whiskyName = whisky.name ?? 'Unbekannt';
    doc.setFontSize(12);
    doc.setTextColor(PRINT_COLORS.black);
    doc.setFont('helvetica', 'italic');
    // Lange Namen umbrechen statt abschneiden
    const nameLines = doc.splitTextToSize(whiskyName, usableWidth - 30);
    doc.text(nameLines, margin + 10, y + 6);

    // Meta-Info
    const metaParts = [
      whisky.distillery,
      whisky.age ? `${whisky.age}y` : null,
      whisky.abv ? `${whisky.abv}%` : null,
      whisky.region,
      whisky.caskType,
    ].filter(Boolean);
    doc.setFontSize(7);
    doc.setTextColor(PRINT_COLORS.gold);
    doc.setFont('helvetica', 'normal');
    doc.text(metaParts.join(' · ').toUpperCase(), margin + 10, y + 11);

    // Flavour-Tags (nur wenn vorhanden)
    const tags = getFlavorTags(whisky);
    if (tags.length > 0) {
      let tagX = margin + 10;
      const tagY = y + 15;
      tags.forEach(tag => {
        const tagWidth = doc.getTextWidth(tag) + 4;
        doc.setDrawColor(PRINT_COLORS.gold);
        doc.setLineWidth(0.3);
        doc.roundedRect(tagX, tagY - 3, tagWidth, 5, 1, 1, 'S');
        doc.setFontSize(6);
        doc.setTextColor(PRINT_COLORS.gold);
        doc.text(tag, tagX + 2, tagY + 0.5);
        tagX += tagWidth + 3;
      });
      y += 7;
    }

    y += 16;

    // Score-Kreise pro Dimension
    DIMS.forEach(dim => {
      doc.setFontSize(7);
      doc.setTextColor(PRINT_COLORS.gold);
      doc.setFont('helvetica', 'normal');
      doc.text(dim.toUpperCase(), margin + 10, y + 3.5);
      drawScoreCircles(doc, margin + 30, y, 4.5);
      y += 7;
    });

    // Notiz-Zeile
    doc.setFontSize(7);
    doc.setTextColor(PRINT_COLORS.textMuted);
    doc.setFont('helvetica', 'italic');
    doc.text('Notiz:', margin + 10, y + 3);
    drawGoldLine(doc, margin + 22, y + 2, usableWidth - 22, 0.3);
    y += 6;

    // Trennlinie zwischen Whiskies
    drawGoldLine(doc, margin, y, usableWidth, 0.2);
    y += 5;
  }

  // ── FOOTER ─────────────────────────────────────────────────────────────
  doc.setFontSize(6);
  doc.setTextColor(PRINT_COLORS.textMuted);
  doc.text('CaskSense', margin, pageHeight - 8);
  doc.text('casksense.com', pageWidth - margin, pageHeight - 8, { align: 'right' });
}
```

## SCHRITT 3 — Tasting-Matte: `drawTastingMat()` ersetzen

```typescript
async function drawTastingMat(
  doc: jsPDF,
  tasting: Tasting,
  whiskies: Whisky[],
  participant: Participant,
  pageWidth: number,
  pageHeight: number,
  margin: number = 15
): Promise<void> {
  const usableWidth = pageWidth - 2 * margin;
  let y = margin;

  // ── HEADER ─────────────────────────────────────────────────────────────
  doc.setFontSize(7);
  doc.setTextColor(PRINT_COLORS.gold);
  doc.text('TASTING-MATTE', margin, y + 3);

  doc.setFontSize(14);
  doc.setTextColor(PRINT_COLORS.black);
  doc.setFont('helvetica', 'italic');
  doc.text(tasting.title, margin, y + 10);

  const dramsLabel = `${whiskies.length} ${whiskies.length === 1 ? 'Dram' : 'Drams'}`;
  const dateStr = tasting.date ? formatDate(tasting.date) : '';
  doc.setFontSize(7);
  doc.setTextColor(PRINT_COLORS.textMuted);
  doc.setFont('helvetica', 'normal');
  doc.text([dramsLabel, dateStr, tasting.hostName].filter(Boolean).join(' · '), margin, y + 15);

  // QR-Code oben rechts
  const qrX = pageWidth - margin - 22;
  await generateParticipantQR(tasting.id, participant.id, doc, qrX, y, 20);

  y += 22;
  drawGoldLine(doc, margin, y, usableWidth, 0.4);
  y += 6;

  // ── SPALTEN-HEADER ──────────────────────────────────────────────────────
  doc.setFontSize(6);
  doc.setTextColor(PRINT_COLORS.gold);
  doc.text('#', margin, y);
  doc.text('STICHWORT', margin + 14, y);
  doc.text('SCORE / 100', usableWidth * 0.55 + margin, y);
  doc.text('FAVORIT', usableWidth * 0.78 + margin, y);
  y += 4;
  drawGoldLine(doc, margin, y, usableWidth, 0.3);
  y += 4;

  // ── WHISKY-ZEILEN ───────────────────────────────────────────────────────
  for (let i = 0; i < whiskies.length; i++) {
    const whisky = whiskies[i];
    const rowY = y;

    // Nummer
    doc.setFontSize(18);
    doc.setTextColor(PRINT_COLORS.gold);
    doc.setFont('helvetica', 'normal');
    doc.text(String(i + 1), margin, rowY + 7);

    // Blind-Name oder echter Name (je nach tasting.isBlind)
    doc.setFontSize(8);
    doc.setTextColor(PRINT_COLORS.textMuted);
    doc.setFont('helvetica', 'italic');
    const displayName = tasting.isBlind ? 'Blind Sample' : (whisky.name ?? '');
    doc.text(displayName, margin + 14, rowY + 4);

    // Stichwort-Linie
    drawGoldLine(doc, margin + 14, rowY + 8, usableWidth * 0.38, 0.4);
    doc.setFontSize(6);
    doc.setTextColor(PRINT_COLORS.textMuted);
    doc.setFont('helvetica', 'italic');
    doc.text('Stichwort', margin + 14, rowY + 7);

    // Score-Feld
    const scoreX = usableWidth * 0.55 + margin;
    drawGoldLine(doc, scoreX, rowY + 8, usableWidth * 0.18, 0.4);
    doc.setFontSize(6);
    doc.setTextColor(PRINT_COLORS.gold);
    doc.text('/ 100', scoreX + usableWidth * 0.19, rowY + 7.5);

    // Favorit-Checkbox
    const cbX = usableWidth * 0.78 + margin;
    doc.setDrawColor(PRINT_COLORS.gold);
    doc.setLineWidth(0.4);
    doc.rect(cbX, rowY + 3, 5, 5, 'S');
    doc.setFontSize(6);
    doc.setTextColor(PRINT_COLORS.textMuted);
    doc.text('Ja', cbX + 7, rowY + 7);

    y += 14;
    drawGoldLine(doc, margin, y, usableWidth, 0.15);
    y += 3;

    // Seitenumbruch
    if (y > pageHeight - 25 && i < whiskies.length - 1) {
      doc.addPage();
      y = margin;
    }
  }

  // ── FOOTER ─────────────────────────────────────────────────────────────
  doc.setFontSize(6);
  doc.setTextColor(PRINT_COLORS.textMuted);
  doc.text('CaskSense', margin, pageHeight - 8);
  doc.text('casksense.com', pageWidth - margin, pageHeight - 8, { align: 'right' });
}
```

### In der bestehenden Batch-Generierungsfunktion:

```typescript
// VORHER: alle Teilnehmer bekommen denselben generischen Bogen

for (const participant of participants) {
```

## Die vier unbrechbaren Regeln

1. **Kein dunkler Hintergrund** — printer-friendly bedeutet weißer Grund.
2. **Whisky-Namen nie abschneiden** — immer `doc.splitTextToSize()`.
3. **Keine Balance-Dimension** — nur Nase · Gaumen · Abgang · Gesamt.
4. **Flavour-Tags nur aus flavorProfile** — nie erfunden, nie geraten.
