### Teaser-Text Logik

```typescript
// Ersten Satz aus dem Markdown-Report extrahieren:
const extractTeaser = (reportMarkdown: string): string => {
  // Markdown-Syntax entfernen
  const plain = reportMarkdown
    .replace(/#{1,6}\s/g, '')
    .replace(/\*\*/g, '')
    .replace(/\*/g, '')
    .replace(/\n/g, ' ')
    .trim();
  
  // Ersten Satz oder max 120 Zeichen
  const firstSentence = plain.split(/[.!?]/)[0];
  if (firstSentence.length <= 120) return firstSentence + '.';
  return plain.substring(0, 120) + '...';
};
```

### Timestamp Logik

```typescript
const formatRelativeTime = (date: string, language: string): string => {
  const diff = Date.now() - new Date(date).getTime();
  const days = Math.floor(diff / 86400000);
  
  if (language.startsWith('de')) {
    if (days === 0) return 'Heute aktualisiert';
    if (days === 1) return 'Vor gestern aktualisiert';
    return `Vor ${days} Tagen aktualisiert`;
  } else {
    if (days === 0) return 'Updated today';
    if (days === 1) return 'Updated yesterday';
    return `Updated ${days} days ago`;
  }
};
```
