  const STORY_AI_RATE_LIMIT = 30;
  const STORY_AI_RATE_WINDOW_MS = 60 * 1000;

  const checkStoryAiRate = (key: string): { allowed: boolean; retryAfterSeconds?: number } => {
    const now = Date.now();
    const entry = storyAiRateLimit.get(key);
    if (!entry || now > entry.resetAt) {
      storyAiRateLimit.set(key, { count: 1, resetAt: now + STORY_AI_RATE_WINDOW_MS });
      return { allowed: true };
    }
    if (entry.count >= STORY_AI_RATE_LIMIT) {
      return { allowed: false, retryAfterSeconds: Math.ceil((entry.resetAt - now) / 1000) };
    }
    entry.count++;
    return { allowed: true };
  };

  // ===== TASTING STORY (Storybuilder block document on tastings.storyBlocks) =====

  const tastingStoryBlocksSchema = z.array(
    z.object({
      id: z.string().min(1).max(128),
      type: z.string().min(1).max(64),
      payload: z.record(z.unknown()),
      hidden: z.boolean().optional(),
    }),
  ).max(500);

  const sanitizeTastingId = (raw: unknown): string => {
    const value = typeof raw === "string" ? raw : Array.isArray(raw) && typeof raw[0] === "string" ? raw[0] : "";
    return value.replace(/[^a-zA-Z0-9_-]/g, "").slice(0, 128);
  };

  const buildTastingStoryMeta = async (tasting: Awaited<ReturnType<typeof storage.getTasting>>) => {
    if (!tasting) return null;
    const created = tasting.createdAt instanceof Date ? tasting.createdAt.toISOString() : (typeof tasting.createdAt === "string" ? tasting.createdAt : null);
    let updated = created;
    const latestVersion = await storage.getLatestStoryVersion("tasting", tasting.id);
    if (latestVersion) {
      const versionAt = latestVersion.createdAt instanceof Date ? latestVersion.createdAt.toISOString() : (typeof latestVersion.createdAt === "string" ? latestVersion.createdAt : null);
      if (versionAt) updated = versionAt;
    }
    return {
      id: tasting.id,
      title: tasting.title,
      hostId: tasting.hostId ?? null,
      hostName: (tasting as { hostName?: string | null }).hostName ?? null,
      scheduledAt: tasting.date ?? null,
      location: tasting.location ?? null,
      coverImageUrl: tasting.coverImageUrl ?? null,
      storyEnabled: !!tasting.storyEnabled,
      createdAt: created,
      updatedAt: updated,
    };
  };

  const loadTastingStoryDocument = async (tastingId: string) => {
    const tasting = await storage.getTasting(tastingId);
    if (!tasting) return { tasting: null as null | Awaited<ReturnType<typeof storage.getTasting>>, blocks: [] as z.infer<typeof tastingStoryBlocksSchema>, autoGenerated: false };
    const raw = tasting.storyBlocks;
    if (raw !== null && raw !== undefined) {
      const parsed = tastingStoryBlocksSchema.safeParse(raw);
      if (parsed.success) {
        return { tasting, blocks: parsed.data, autoGenerated: false };
      }
    }
    const { buildInitialTastingStoryBlocks } = await import("./tastingStoryAutoFill");
    const [whiskies, participantsRaw, ratings] = await Promise.all([
      storage.getWhiskiesForTasting(tastingId),
      storage.getTastingParticipants(tastingId),
      storage.getRatingsForTasting(tastingId),
    ]);
    const participants = participantsRaw
      .map((row) => row.participant)
      .filter((p): p is NonNullable<typeof p> => !!p)
      .map((p) => ({ id: p.id, displayName: (p as { displayName?: string | null }).displayName ?? null, name: p.name ?? null }));
    const initialBlocks = buildInitialTastingStoryBlocks({
      tasting,
      whiskies,
      participantCount: participantsRaw.length,
      participants,
      ratings,
    });
    const reparsed = tastingStoryBlocksSchema.safeParse(initialBlocks);
    if (!reparsed.success) {
      return { tasting, blocks: [] as z.infer<typeof tastingStoryBlocksSchema>, autoGenerated: false };
    }
    await storage.updateTasting(tastingId, { storyBlocks: reparsed.data });
    const refreshed = await storage.getTasting(tastingId);
    return { tasting: refreshed ?? tasting, blocks: reparsed.data, autoGenerated: true };
  };

  const checkTastingStoryAccess = async (
    req: Request,
    tastingId: string,
  ): Promise<
    | { ok: true; participant: Participant; canEdit: boolean }
    | { ok: false; status: number; message: string }
  > => {
    const auth = await requireAuth(req);
    if (!auth.authenticated) return { ok: false, status: auth.status, message: auth.message };
    const tasting = await storage.getTasting(tastingId);
    if (!tasting) return { ok: false, status: 404, message: "Verkostung nicht gefunden" };
    const isAdmin = auth.participant.role === "admin";
    const isHost = tasting.hostId === auth.participant.id;
    if (!isAdmin && !isHost) {
      return { ok: false, status: 403, message: "Nur Host oder Admin haben Zugriff" };
    }
    return { ok: true, participant: auth.participant, canEdit: true };
  };

  app.get("/api/tasting-stories/:tastingId", async (req: Request, res: Response) => {
    try {
      const tastingId = sanitizeTastingId(req.params.tastingId);
      if (!tastingId) return res.status(400).json({ message: "tastingId fehlt" });
      const access = await checkTastingStoryAccess(req, tastingId);
      if (!access.ok) return res.status(access.status).json({ message: access.message });
      const result = await loadTastingStoryDocument(tastingId);
      if (!result.tasting) return res.status(404).json({ message: "Verkostung nicht gefunden" });
      const meta = await buildTastingStoryMeta(result.tasting);
      const updatedAt = meta?.updatedAt ?? new Date().toISOString();
      return res.json({
        tasting: meta,
        document: {
          schemaVersion: 1,
          theme: "casksense-editorial",
          blocks: result.blocks,
          metadata: { createdAt: updatedAt, updatedAt, title: result.tasting.title },
        },
        autoGenerated: result.autoGenerated,
        canEdit: access.canEdit,
      });
    } catch (e: unknown) {
      console.error("[tasting-stories/get] error:", e);
      const msg = e instanceof Error ? e.message : "Laden fehlgeschlagen";
      return res.status(500).json({ message: msg });
    }
  });

  app.put("/api/tasting-stories/:tastingId", async (req: Request, res: Response) => {
    try {

      if (!access.ok) return res.status(access.status).json({ message: access.message });
      const body = z.object({ blocks: tastingStoryBlocksSchema }).safeParse(req.body);
      if (!body.success) {
        return res.status(422).json({ message: "Ungültige Blockdaten" });
      }
      await storage.updateTasting(tastingId, { storyBlocks: body.data.blocks });
      try {
        await storage.saveStoryVersion({
          sourceType: "tasting",
          sourceId: tastingId,
          blocksJson: body.data.blocks,
          isAuto: true,
          name: null,
          createdById: access.participant.id,
        });
        await storage.pruneOldAutoStoryVersions("tasting", tastingId, 20);
      } catch (versionErr) {
        console.warn("[tasting-stories/put] version save warning:", versionErr);
      }
      return res.json({ ok: true, updatedAt: new Date().toISOString() });
    } catch (e: unknown) {
      console.error("[tasting-stories/put] error:", e);
      const msg = e instanceof Error ? e.message : "Speichern fehlgeschlagen";
      return res.status(500).json({ message: msg });
    }
  });

  app.get("/api/public/tasting-stories/:tastingId", async (req: Request, res: Response) => {
    try {
      const tastingId = sanitizeTastingId(req.params.tastingId);
      if (!tastingId) return res.status(400).json({ message: "tastingId fehlt" });
      const tasting = await storage.getTasting(tastingId);
      if (!tasting) return res.status(404).json({ message: "Verkostung nicht gefunden" });
      const auth = await requireAuth(req);
      const isAdmin = auth.authenticated && auth.participant.role === "admin";
      const isHost = auth.authenticated && tasting.hostId === auth.participant.id;
      if (!isAdmin && !isHost) {
        if (!tasting.storyEnabled) {
          return res.status(403).json({ message: "Story noch nicht freigegeben" });
        }
        if (!auth.authenticated) {
          return res.status(401).json({ message: "Anmeldung erforderlich" });
        }
        const isMember = await storage.isParticipantInTasting(tastingId, auth.participant.id);
        if (!isMember) {
          return res.status(403).json({ message: "Zugriff nur fuer eingeladene Teilnehmer" });
        }
      }
      const parsed = tastingStoryBlocksSchema.safeParse(tasting.storyBlocks);
      const blocks = parsed.success ? parsed.data : [];
      const meta = await buildTastingStoryMeta(tasting);
      const updatedAt = meta?.updatedAt ?? new Date().toISOString();
      return res.json({
        tasting: meta,
        document: {
          schemaVersion: 1,
          theme: "casksense-editorial",
          blocks,
          metadata: { createdAt: updatedAt, updatedAt, title: tasting.title },
        },
        autoGenerated: false,
        canEdit: isAdmin || isHost,
      });
    } catch (e: unknown) {
      console.error("[tasting-stories/public] error:", e);
      const msg = e instanceof Error ? e.message : "Laden fehlgeschlagen";
      return res.status(500).json({ message: msg });
    }
  });

  // Serve the cinematic tasting story standalone page (block-renderer first, legacy template fallback)
  app.get("/tasting-story/:id", async (req: Request, res: Response) => {
    const safeId = req.params.id.replace(/[^a-zA-Z0-9_-]/g, "");
    const forceLegacy = req.query.legacy === "1";
    if (!forceLegacy && safeId) {
      try {
        const tasting = await storage.getTasting(safeId);
        const parsed = tasting ? tastingStoryBlocksSchema.safeParse(tasting.storyBlocks) : null;
        if (tasting && parsed && parsed.success && parsed.data.length > 0) {
          const indexCandidates = [
            path.resolve(process.cwd(), "dist", "public", "index.html"),
            path.resolve(process.cwd(), "client", "index.html"),
          ];
          const indexPath = indexCandidates.find((p) => fs.existsSync(p));
          if (indexPath) {
            const html = fs.readFileSync(indexPath, "utf-8");
            res.setHeader("Content-Type", "text/html; charset=utf-8");
            res.setHeader("Cache-Control", "no-cache, no-store, must-revalidate");
            return res.send(html);
          }
        }
      } catch (preflightErr) {
        console.warn("[tasting-story] preflight failed, falling back:", preflightErr);
      }
    }
    const templatePath = path.join(process.cwd(), "client", "public", "tasting-story", "template.html");
    try {
      let html = fs.readFileSync(templatePath, "utf-8");
      html = html.replace("__TASTING_ID__", safeId);
      res.setHeader("Content-Type", "text/html; charset=utf-8");
      res.setHeader("Cache-Control", "no-cache, no-store, must-revalidate");
      res.send(html);
    } catch (e: unknown) {
      const msg = e instanceof Error ? e.message : String(e);
      console.error("tasting-story template read error:", msg);
      res.status(500).send("Story page template not found.");
    }
  });

  return httpServer;
